* Fix screen tearing

For amd based graphics
#+begin_src conf
   Section "Device"

     Identifier "AMD Graphics"
     Driver "amdgpu"
     Option "TearFree" "true"

  EndSection
#+end_src

For intel based graphics
#+begin_src conf
   Section "Device"

     Identifier "Intel Graphics"
     Driver "intel"
     Option "TearFree" "true"

  EndSection
#+end_src
