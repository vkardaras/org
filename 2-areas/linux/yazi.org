** Yazi core features
:PROPERTIES:
:CUSTOM_ID: yazi-core-features
:END:
#+begin_src sh
# run yazi
yazi

# view help menu
~ or F1

# open a hew tab
t

# switch tabs
number keys or []

# select file
space

# enter visual mode
v

# copy file
y

# paste file
p

# rename file
r

# delete file
d

# delete file permanently
D

# create a new file or directory
a
#+end_src

w

: or ;

# edit the file '.config/yazi/theme.toml'
nvim ~/.config/yazi/theme.toml

[flavor]

light = "tokionight-night"
