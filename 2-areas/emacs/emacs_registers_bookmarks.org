#+title: Emacs Registers/Bookmarks

* Registers

#+begin_src shell
  # store a point to a register
  C-x r SPC

  # jump to register
  C-x r j

  # store text to register
  C-x r s

  # store text to register and cut it
  C-u C-x r s

  # insert text from register
  C-x r i

  # append text to register
  C-x r +

  # store window configuration to register
  C-x r w

  # store a file location to a register
  (set-register ?l '(file . "/tmp/temp_file.txt"))

#+end_src
