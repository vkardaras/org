* Doom Emacs
:PROPERTIES:
:CUSTOM_ID: doom-emacs
:END:
** Run Emacs

#+begin_src sh
# add emacs to PATH
export PATH="$HOME/.config/emacs/bin:$PATH"

# reload emacs
SPC h r r
#+end_src

** Emacs general

#+begin_src sh
# zoom in
C-x C+

# zoom out
C-x C-

# open file
SPC .

# create or open bookmark
SPC RET

# open a project
SPC p p
#+end_src

** Projects with Projectile, File Explorer with Treemacs & EShel

#+begin_src sh
# discover projects in directory
M-x projecttile-discover-projects-in-directory

# select project
SPC p p

# find files in configuration
SPC f p

# open shell
SPC o e

# list recent files
SPC f r
#+end_src
